sp.ovlp <- fi_den%>%
  gather(c("HS", "HC"), key = sp, value = CPUE)%>%
  gather(c("GillNet", "Trawler"), key = gear, value = intensity)%>%
  group_by(gear, sp)%>%
  summarise(overlap = sum(CPUE > 0 & intensity > 0),
            extent.sp = sum(CPUE>0),
